<script>
    // Simple Language Logic
    let currentLang = 'en';
    function toggleLang() {
        currentLang = currentLang === 'en' ? 'hi' : 'en';
        document.getElementById('hero-title').innerText = currentLang === 'en' ? 
            'Know your legal rights in simple language.' : 
            'अपनी कानूनी अधिकारों को सरल भाषा में जानें।';
        document.getElementById('userInput').placeholder = currentLang === 'en' ? 
            'Example: My landlord is not returning my deposit...' : 
            'उदाहरण: मेरा मकान मालिक मेरी सुरक्षा राशि वापस नहीं कर रहा है...';
    }

    // Mock AI Logic
    function analyzeProblem() {
        const input = document.getElementById('userInput').value.toLowerCase();
        const resultsArea = document.getElementById('results-area');
        const resTitle = document.getElementById('result-type');
        const resExpl = document.getElementById('result-explanation');
        const resSteps = document.getElementById('action-steps');

        if(input.length < 5) {
            alert("Please describe your situation in more detail.");
            return;
        }

        resultsArea.style.display = 'block';
        resSteps.innerHTML = ''; // Clear previous

        // Mock Intent Classification
        if (input.includes('landlord') || input.includes('rent') || input.includes('deposit')) {
            resTitle.innerText = "🏠 Tenant Rights (Rent Control)";
            resExpl.innerText = "Under local Rent Control laws, a landlord cannot withhold a security deposit without providing a list of specific damages.";
            const steps = ["Send a formal written notice via WhatsApp or Speed Post.", "Keep proof of all rent receipts.", "Approach the Rent Controller office if they don't reply in 15 days."];
            steps.forEach(s => resSteps.innerHTML += `<li>${s}</li>`);
        } 
        else if (input.includes('salary') || input.includes('boss') || input.includes('fired')) {
            resTitle.innerText = "👷 Worker Rights";
            resExpl.innerText = "The Payment of Wages Act ensures you are paid on time. Termination requires a notice period or pay-in-lieu.";
            const steps = ["Gather your offer letter and bank statements.", "Contact your HR in writing.", "Visit the local Labour Commissioner's office."];
            steps.forEach(s => resSteps.innerHTML += `<li>${s}</li>`);
        }
        else {
            resTitle.innerText = "⚖️ General Legal Help";
            resExpl.innerText = "Based on your description, you should consult a free legal aid volunteer to categorize this specific issue.";
            const steps = ["Visit the nearest Legal Aid Center (shown below).", "Carry a valid ID proof.", "Ask for a 'Para-legal Volunteer'."];
            steps.forEach(s => resSteps.innerHTML += `<li>${s}</li>`);
        }

        // Auto-scroll to results
        resultsArea.scrollIntoView({ behavior: 'smooth' });
    }
</script>
